NSObject的底层实现

@interface NSObject {
Class isa;
}
@end

编译成结构体：

struct NSObject_IMPL {
Class isa;
}
typedef struct objc_class *Class;
NSObject *obj = [[NSObject alloc] init];

OC对象的本质

@interface Student : NSObject {
@public
int _no;
int _ago;
}
@end

编译成结构体：

struct Student_IMPL {
Class isa;
int _no;
int _age;
}

举例：

Student *stu = [[Student alloc] init];
stu->_no = 4;
stu->_age = 5;
